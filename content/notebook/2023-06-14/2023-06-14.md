Some code for running a secure multi-party vote, there aren't a whole lot of mpyc examples out on the internet right now.

```
from mpyc.runtime import mpc

def elect(secintf, candidates, myvote):
    """
    Hold a secure multi-party election.

    Args:
        secint: Factory function for creating mpyc secure integers
        candidates: How many candidates are running as a public int
        myvote: Candidate receiving this party's vote as a mpyc secure integer
            from 0 to candidates-1

    Returns:
        Winning candidate as a mpyc secure integer from 0 to candidates-1
    """

    # Votes from all parties in a list
    votes = mpc.input(myvote)

    # Secure list for counting the votes for each candidate
    counts = mpc.seclist([0] * candidates, secintf)

    for vote in votes:
        counts[vote] += mpc.and_(vote >= 0, vote < candidates)

    # No longer need to hide which candidates are being incremented
    counts = list(counts)

    # Index (candidate) with the highest vote
    return mpc.argmax(counts)[0]

VOTES = [1, 0, 0, 4, 4, 4]
secret_factory = mpc.SecInt(32)

mpc.run(mpc.start())

my_vote = secret_factory(VOTES[mpc.pid])
output = mpc.output(elect(secret_factory, 2, my_vote))

mpc.run(mpc.shutdown())

print(f'P{mpc.pid} reports winning index {output.result()}')
```
